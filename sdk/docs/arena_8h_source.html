7// arena.h - Arena memory allocator
8//
9// Arenas provide fast, zero-fragmentation memory allocation with bulk free.
10// Allocations bump a pointer forward; freeing resets the pointer.
11//
12// Three standard arenas are provided:
13// ng_arena_persistent - Lives entire game (player data, global state)
14// ng_arena_state - Cleared on level/screen changes
15// ng_arena_frame - Cleared every frame (temp strings, scratch)
16
17#ifndef ARENA_H
18#define ARENA_H
19
20#include <types.h>
21
22// Arena structure
23typedef struct NGArena {
24 u8 *base; // Start of memory region
25 u8 *current; // Current allocation pointer (bump pointer)
26 u8 *end; // End of memory region
27} NGArena;
28
29// Mark for save/restore (temporary allocations)
30typedef u8 *NGArenaMark;
31
32// === Core Operations ===
33
34// Initialize arena with external buffer
35void NGArenaInit(NGArena *arena, void *buffer, u32 size);
36
37// Allocate size bytes (4-byte aligned), returns NULL if out of memory
38void *NGArenaAlloc(NGArena *arena, u32 size);
39
40// Reset arena to empty (instant bulk free)
41void NGArenaReset(NGArena *arena);
42
43// === Temporary Allocations ===
44
45// Save current position (for temporary allocations)
46NGArenaMark NGArenaSave(NGArena *arena);
47
48// Restore to saved position (frees everything allocated since save)
49void NGArenaRestore(NGArena *arena, NGArenaMark mark);
50
51// === Queries ===
52
53// Get bytes currently used
54u32 NGArenaUsed(NGArena *arena);
55
56// Get bytes remaining
57u32 NGArenaRemaining(NGArena *arena);
58
59// === Convenience Macros ===
60
61// Allocate a single struct
62#define NG_ARENA_ALLOC(arena, type) \
63 ((type *)NGArenaAlloc((arena), sizeof(type)))
64
65// Allocate an array of structs
66#define NG_ARENA_ALLOC_ARRAY(arena, type, count) \
67 ((type *)NGArenaAlloc((arena), sizeof(type) * (count)))
68
69// === Standard Arenas ===
70
71// Default arena sizes (can be overridden before including arena.h)
72#ifndef NG_ARENA_PERSISTENT_SIZE
73#define NG_ARENA_PERSISTENT_SIZE 8192 // 8 KB
74#endif
75
76#ifndef NG_ARENA_STATE_SIZE
77#define NG_ARENA_STATE_SIZE 24576 // 24 KB
78#endif
79
80#ifndef NG_ARENA_FRAME_SIZE
81#define NG_ARENA_FRAME_SIZE 4096 // 4 KB
82#endif
83
84// Standard arenas (initialized by NGArenaSystemInit)
85extern NGArena ng_arena_persistent;
86extern NGArena ng_arena_state;
87extern NGArena ng_arena_frame;
88
89// Initialize standard arenas (call once at startup)
90void NGArenaSystemInit(void);
91
92#endif // ARENA_H
