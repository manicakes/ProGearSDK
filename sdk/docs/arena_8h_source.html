7// arena.h - Arena memory allocator
8//
9// Arenas provide fast, zero-fragmentation memory allocation with bulk free.
10// Allocations bump a pointer forward; freeing resets the pointer.
11//
12// Three standard arenas are provided:
13// ng_arena_persistent - Lives entire game (player data, global state)
14// ng_arena_state - Cleared on level/screen changes
15// ng_arena_frame - Cleared every frame (temp strings, scratch)
