7// spring.h - Spring physics animation system
8//
9// Springs provide smooth, natural-feeling animations with overshoot and settling.
10// Use for UI elements, camera movement, actor motion, or any value that should
11// animate smoothly toward a target.
12//
13// The spring physics model:
14// acceleration = -stiffness * (value - target) - damping * velocity
15//
16// Higher stiffness = faster response
17// Higher damping = less overshoot (critical damping = no overshoot)
18
19#ifndef SPRING_H
20#define SPRING_H
21
22#include <types.h>
23#include <ngmath.h>
24
25// === Spring Presets ===
26// These provide good defaults for common use cases.
27// Format: { stiffness, damping }
28// Tuned for 60fps, settling in ~200-300ms (12-18 frames)
29
30// Snappy with minimal overshoot - good for UI cursors
31#define NG_SPRING_SNAPPY_STIFFNESS FIX_FROM_FLOAT(0.35)
32#define NG_SPRING_SNAPPY_DAMPING FIX_FROM_FLOAT(0.65)
33
34// Bouncy with noticeable overshoot - good for playful UI
35#define NG_SPRING_BOUNCY_STIFFNESS FIX_FROM_FLOAT(0.25)
36#define NG_SPRING_BOUNCY_DAMPING FIX_FROM_FLOAT(0.45)
37
38// Smooth with no overshoot - good for subtle transitions
39#define NG_SPRING_SMOOTH_STIFFNESS FIX_FROM_FLOAT(0.20)
40#define NG_SPRING_SMOOTH_DAMPING FIX_FROM_FLOAT(0.80)
41
42// Quick snap - very fast, almost instant
43#define NG_SPRING_QUICK_STIFFNESS FIX_FROM_FLOAT(0.50)
44#define NG_SPRING_QUICK_DAMPING FIX_FROM_FLOAT(0.70)
45
46// === 1D Spring ===
47
48typedef struct NGSpring {
49 fixed value; // Current animated value
50 fixed velocity; // Current velocity
51 fixed target; // Target value to animate toward
52 fixed stiffness; // Spring constant (higher = faster)
53 fixed damping; // Damping ratio (higher = less bouncy)
54} NGSpring;
55
56// Initialize spring with starting value and default snappy physics
57void NGSpringInit(NGSpring *spring, fixed initial);
58
59// Initialize spring with custom physics parameters
60void NGSpringInitEx(NGSpring *spring, fixed initial, fixed stiffness, fixed damping);
61
62// Set target value (spring will animate toward it)
63void NGSpringSetTarget(NGSpring *spring, fixed target);
64
65// Snap immediately to value (no animation)
66void NGSpringSnap(NGSpring *spring, fixed value);
67
68// Add impulse to velocity (for responsive input feel)
69void NGSpringImpulse(NGSpring *spring, fixed impulse);
70
71// Update spring physics (call once per frame)
72void NGSpringUpdate(NGSpring *spring);

74// Get current value
75static inline fixed NGSpringGet(NGSpring *spring) {
76 return spring->value;
77}
78
79// Get current value as integer

84// Check if spring has settled (near target with low velocity)
