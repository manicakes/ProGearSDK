85static inline s16 NGSpriteScreenToHardwareY(s16 screen_y) {
86 s16 y_val = (s16)(496 - screen_y);
87 if (y_val < 0)
88 y_val = (s16)(y_val + 512);
89 return (s16)(y_val & 0x1FF);
90}

103static inline u8 NGSpriteAdjustedHeight(u8 rows, u8 v_shrink) {
104 /* Ceiling division: (rows * v_shrink + 254) / 255 */
105 u16 adjusted = (u16)(((u16)rows * v_shrink + 254) / 255);
106 if (adjusted < 1)
107 adjusted = 1;
108 if (adjusted > NG_SPRITE_MAX_HEIGHT)
109 adjusted = NG_SPRITE_MAX_HEIGHT;
110 return (u8)adjusted;
111}

122static inline u16 NGSpriteSCB3(s16 screen_y, u8 height) {
123 s16 hw_y = NGSpriteScreenToHardwareY(screen_y);
124 return (u16)(((u16)hw_y << 7) | (height & 0x3F));
125}

160static inline void NGSpriteHideRange(u16 first_sprite, u8 count) {
161 if (count == 0)
162 return;
163 NG_VRAM_DECLARE_BASE();
164 NG_VRAM_SETUP_FAST(NG_SCB3_BASE + first_sprite, 1);
165 NG_VRAM_CLEAR_FAST(count);
166}
