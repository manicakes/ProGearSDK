63static inline fixed FIX_MUL(fixed a, fixed b) {
64 s16 a_hi = (s16)(a >> 16);
65 u16 a_lo = (u16)a;
66 s16 b_hi = (s16)(b >> 16);
67 u16 b_lo = (u16)b;
68 s32 result = ((s32)a_hi * b_hi) << 16;
69 result += (s32)a_hi * b_lo;
70 result += (s32)a_lo * b_hi;
71 result += ((u32)a_lo * b_lo) >> 16;
72 return result;
73}
